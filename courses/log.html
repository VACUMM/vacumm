# -*- coding: utf8 -*-
"""Passer par un logger"""

from vacumm.misc.log import Logger # logger avancé
#from vacumm.misc.io import Logger # logger plus simple

# Initialisation

    format='[%(asctime)s %(name)s %(levelname)s] %(message)s')
